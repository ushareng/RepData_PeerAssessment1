The strategy for filling in all of the missing values in the dataset is replacing the NAs with the mean for that 5-minute interval. Function replaceNA() replaces NA's with the mean for that 5-minute interval and the final result is stored in newData .

replaceNA <- function(steps, interval) {
    replaceData <- NA
    if (!is.na(steps)) 
        replaceData <- c(steps) else replaceData <- (averageSteps[averageSteps$interval == interval, "steps"])
    return(replaceData)
}
newData <- activity
newData$steps <- mapply(replaceNA, newData$steps, newData$interval)
